create table customers(
c_id int,
c_name varchar(32) NOT NULL,
c_address varchar(32) NOT NULL,
email varchar(50) NOT NULL,
mobile varchar(32) NOT NULL,
PRIMARY KEY(c_id)
);

order_id int,

FOREIGN KEY(cust_id) REFERENCES customers(c_id)
);
